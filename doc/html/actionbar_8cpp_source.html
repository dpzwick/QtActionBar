106 void ActionBar::adjustContent() {
107  // Get size of one em (text height in pixels)
108  int em=fontMetrics().height();
109 
110  viewMenu->repaint();
111 
112  // update size of app icon and overflow menu button
113  appIcon->setIconSize(QSize(2*em,2*em));
114  overflowButton->setIconSize(QSize(2*em,2*em));
115 
116  // Check if all action buttons fit into the available space with text beside icon.
117  bool needOverflow=false;
118  int space=width() - appIcon->sizeHint().width() - viewControl->sizeHint().width();
119  for (int i=0; i<buttonActions.size(); i++) {
120  QAction* action=buttonActions.at(i);
121  QToolButton* button=actionButtons.at(i);
122  if (action->isVisible()) {
123  button->setIconSize(QSize(2*em,2*em));
124  button->setToolButtonStyle(Qt::ToolButtonTextBesideIcon);
125  space-=button->sizeHint().width();
126  }
127  }
128  if (space<0) {
129  // Not enough space.
130  // Check if all action buttons fit into the available space without text.
131  int space=width() - appIcon->sizeHint().width() - viewControl->sizeHint().width();
132  for (int i=0; i<buttonActions.size(); i++) {
133  QAction* action=buttonActions.at(i);
134  QToolButton* button=actionButtons.at(i);
135  if (action->isVisible()) {
136  button->setToolButtonStyle(Qt::ToolButtonIconOnly);
137  space-=button->sizeHint().width();
138  }
139  }
140  if (space<0) {
141  // The buttons still don't fit, we need an overflow menu.
142  needOverflow=true;
143  }
144  }
145 
146  // Calculate space available to display action buttons
147  overflowMenu->clear();
148  space=width() - appIcon->sizeHint().width() - viewControl->sizeHint().width();
149  if (needOverflow) {
150  space-=overflowButton->sizeHint().width();
151  }
152 
153  // Show/Hide action buttons and overflow menu entries
154  for (int i=0; i<buttonActions.size(); i++) {
155  QAction* action=buttonActions.at(i);
156  QToolButton* button=actionButtons.at(i);
157  if (action->isVisible()) {
158  space-=button->sizeHint().width();
159  if (space>=0) {
160  // show as button
161  button->setDisabled(!buttonActions.at(i)->isEnabled());
162  button->show();
163  }
164  else {

168  }
169  }
