86 void MainWindow::keyPressEvent(QKeyEvent *ev) {
87  int key = ev->key();
88 
89  // The menu key has no function by default.
90  // But Android users expect that the menu button opens the overflow menu.
91  // This works only in QT >= 5.4.0, see QTBUG-33276.
92  if (key == Qt::Key_Menu) {
93  qDebug("Catched menu button event");
94  ev->accept();
95  actionBar->openOverflowMenu();
96  return;
97  }
98 
99  // The Qt Framework closes the app when the back button is pressed.
100  // But Android users expect that the back button navigates to the previous page.
101  if (key == Qt::Key_Back && !isMainPage) {
102  qDebug("Catched back button event");
103  ev->accept();
104  goToMain();
105  return;
106  }
107 
108  // Let QWidget process all other keys
109  QWidget::keyPressEvent(ev);
110 }

180 float MainWindow::getPointSize(const QFont& font) {
181  // Workaround against QTBUG-4958
182  // QFontInfo can properly convert from pixels to points but returns
183  // wrong value if the font has already a size in points.
184  if (font.pixelSize() != -1) {
185  // Size is set in pixels, convert to points
186  float pointSize=QFontInfo(font).pointSizeF();
187  qDebug("Converting pixelSize %i to pointSize %f",font.pixelSize(),pointSize);
188  return pointSize;
189  }
190  else {
191  // Size is set in points, no conversion needed
192  return font.pointSizeF();
193  }
194 }
